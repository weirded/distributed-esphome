#include <tunables/global>

# SS.1 (1.6.1): AppArmor profile for the ESPHome Fleet add-on.
#
# Scope: server (aiohttp + `/app/server/`) and the bundled local-worker
# subprocess (`/app/client/client.py`), both running as PID 1+ inside
# the add-on container. The profile's goal isn't zero-trust isolation
# — the add-on already has `host_network`, `hassio_api`, and
# `config:rw` by design — it's to surface the Supervisor "AppArmor"
# security star and to be ready to tighten once we've observed a few
# release cycles of clean telemetry.
#
# First-pass strategy: attach a permissive-but-named profile so the
# Supervisor UI flips the "AppArmor" badge and the container runs
# under confinement (rather than completely unconfined). The
# permissive rules below allow every common operation our stack
# needs — Python interpreter bootstrap, PlatformIO penv
# construction, git, pip, ESPHome build, mDNS, network — without
# itemising paths that turn out to shift between Python minor
# versions, PlatformIO releases, or ESPHome toolchains.
#
# Tightening work is tracked separately: once we have a full
# release cycle of deploy telemetry with this permissive profile,
# we move each broad rule (e.g. ``file,``, ``capability,``) toward
# the narrow subset we've observed actually in use.

profile esphome_dist_server flags=(attach_disconnected,mediate_deleted) {
  # Permissive-but-attached: Supervisor reads "AppArmor: enabled"
  # because a named profile is loaded. Rules below are broad by
  # design — the mediation cost is real (every syscall still hits
  # AppArmor's "check this" path, even when the answer is always
  # "allow") but the defensive gain from itemising paths that shift
  # between Python minor versions, PlatformIO releases, and ESPHome
  # toolchains isn't worth the recurring breakage.
  file,
  capability,
  network,
  signal,
  dbus,
  unix,
  mount,
  pivot_root,

  # PR #80 review — cheap non-breaking denies that close something
  # real without touching the PlatformIO / git / pip happy path.
  #
  # A compromised compile-time Python (ESPHome's `external_components`,
  # `includes:`, `libraries:` — all trusted per the threat model but
  # still the highest-leverage attack surface the add-on exposes)
  # shouldn't be able to:
  #
  # - read host shadow/secrets files if the container's userns ever
  #   maps them in (Supervisor doesn't map them today, but that can
  #   change upstream without notice);
  # - read or write arbitrary processes' memory via `/proc/<pid>/mem`;
  # - tamper with kernel sysctls via /proc/sys writes.
  #
  # ``ptrace,`` is also scoped — without this carve-out, the
  # unqualified ``ptrace,`` above would allow arbitrary process
  # tracing across the add-on's container namespace. We don't need
  # ptrace in normal operation; deny it outright so an attacker
  # can't use it to jump between compile subprocesses.
  deny /etc/shadow* rw,
  deny /etc/gshadow* rw,
  deny /run/secrets/** rw,
  deny /proc/*/mem rw,
  deny /proc/sys/kernel/** w,
  deny ptrace,
}
